HubSpot Integration

const aivory = require('@aivory/sdk');

aivory.integrations.connect({
  provider: 'hubspot',
  apiKey: process.env.HUBSPOT_API_KEY,
  portalId: 'YOUR_PORTAL_ID'
});

// Sync contacts automatically
aivory.workflows.create({
  trigger: 'contact_created',
  actions: ['enrich_data', 'send_welcome_email']
});
